js
複製
編輯
/* 清空原答案區先置入文字 */
a.textContent = item.answer;

/* 如果有 ansphoto (逗號分隔多圖) */
if(item.ansphoto){
  item.ansphoto.split(',').forEach(p=>{
    const img = document.createElement('img');
    img.src = `FAQphoto/${p.trim()}`;
    img.style = "max-width:100%;margin-top:.8rem;border-radius:6px";
    a.appendChild(img);
  });
}

/* 如果有 link 與 link_str */
if(item.link && item.link_str){
  item.link.forEach((url,idx)=>{
    if(!url) return;
    const anchor = document.createElement('a');
    anchor.href = url;
    anchor.textContent = item.link_str[idx] || `參考${idx+1}`;
    anchor.target = "_blank";
    anchor.style = "color:#8A1C1C;text-decoration:underline;margin-left:.4rem";
    a.appendChild(anchor);
  });
}
備註
• ansphoto 只放檔名即可（如 step1.png），程式會自動補前綴 FAQphoto/
• link 和 link_str 長度需相同；若 link_str 未給則顯示「參考1、參考2…」
